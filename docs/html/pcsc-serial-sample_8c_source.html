89void sample(void)
90{
91 BYTE slotCount;
92 BOOL cardPresent[CCID_MAX_SLOT_COUNT] = { 0 };
93
94 /* Verify we can ping the device */
95 if (!ping_device())
96 return;
97
98 /* Retrieve and print the device descriptor */
99 dump_device_descriptor();
100 if (!SCARD_LIB(IsValid)())
101 return;
102
103 /* Retrieve and print the configuration descriptor */
104 dump_configuration_descriptor();
105 if (!SCARD_LIB(IsValid)())
106 return;
107
108 /* Retrieve and print all the string descriptors */
109 dump_string_descriptors();
110 if (!SCARD_LIB(IsValid)())
111 return;
112
113 /* Retrieve and print the interface descriptor */
114 dump_interface_descriptor();
115 if (!SCARD_LIB(IsValid)())
116 return;
117
118 /* Activate the configuration */
119 printf("Starting PC/SC...\n");
120 if (!start_pcsc())
121 return;
122
123 /* Verify we still can ping the device */
124 if (!ping_device())
125 return;
126
127 /* Get the slot count */
128 printf("Reading slot count\n");
129 slotCount = get_slot_count();
130 if (slotCount == 0)
131 return;
132
133 printf("Device has %d slot(s)\n", slotCount);
134 if (slotCount > CCID_MAX_SLOT_COUNT)
135 {
136 printf("WARNING: device has %d card slots, but we can support only %d\n", slotCount, CCID_MAX_SLOT_COUNT);
137 slotCount = CCID_MAX_SLOT_COUNT;
138 }
139
140 if (fTestEchoControl)
141 {
142 printf("Please wait during the test procedure (echo over SCardControl)...\n");
143 if (!test_echo_control())
144 {
145 printf("Test failed!\n");
146 return;
147 }
148 }
149
150 printf("Insert a card in any slot...\n");
151
152 /* Run while the device is up and running */
153 while (SCARD_LIB(IsValid)())
154 {
155 BYTE slot;
156
157 if (fCcidUseNotifications)
158 {
159 /* Wait until a notification arrives */
160 printf("Waiting for a change");
161 wait_status_change((DWORD)-1); /* Use -1 for INFINITE timeout */
162 if (!SCARD_LIB(IsValid)())
163 return;
164
165 /* Verify we still can ping the device */
166 if (!ping_device())
167 return;
168 }
169
170 /* Find the slot(s) were a card has been inserted */
171 for (slot = 0; slot < slotCount; slot++)
172 {
173 /* Ignore slots we can't support */
174 if (slot >= CCID_MAX_SLOT_COUNT)
175 continue;
176
177 if (card_is_present(slot))
178 {
179 if (!cardPresent[slot])
180 {
181 /* A card has been inserted in this slot */
182 cardPresent[slot] = TRUE;
183 printf("Card inserted in slot %d\n", slot);
184
185 /* Run sample over this card */
186 sample_on_slot(slot);
187
188 printf("Transaction with card in slot %d terminated\n", slot);
189 }
190 }
191 else
192 {
193 if (cardPresent[slot])
194 {
195 /* The card has been removed from this slot */
196 cardPresent[slot] = FALSE;
197 printf("Card removed from slot %d\n", slot);
198 }
199 }
200 }
201 }
202
203 /* In case we can exit... */
204 printf("Stopping PC/SC...\n");
205 stop_pcsc();
206}

208static void sample_on_slot(BYTE slot)
209{
210 LONG rc;
211 DWORD dwRecvLength = sizeof(abInOutBuffer);
212
213 /* Try and connect to the card in the slot */
214 rc = SCARD_LIB(Connect)(slot, abInOutBuffer, &dwRecvLength);
215 if (rc != SCARD_ERR(S_SUCCESS))
216 {
217 printf("Failed to connect to the card (rc=%lX)\n", rc);
218 return;
219 }
220
221 /* We've got a card */
222 printf("Card connected, ATR=");
223 for (DWORD i = 0; i < dwRecvLength; i++)
224 printf("%02X", abInOutBuffer[i]);
225 printf("\n");
226
227 /* We use the GET DATA ( GET UID ) command APDU defined by PC/SC v2 chapter 3 for contactless cards */
228 dwRecvLength = sizeof(abInOutBuffer);
229 rc = SCARD_LIB(Transmit)(slot, APDU_GET_UID, sizeof(APDU_GET_UID), abInOutBuffer, &dwRecvLength);
230 if (rc != SCARD_ERR(S_SUCCESS))
231 {
232 printf("Failed to retrieve the UID (rc=%lX)\n", rc);
233 /* This is not always a fatal error */
234 if (SCARD_LIB(IsFatalError)(rc))
235 return;
236 }
237 else
238 {
239 WORD sw = get_status_word(abInOutBuffer, dwRecvLength);
240
241 if (sw != 0x9000)
242 {
243 printf("Failed to retrieve the UID (SW=%04X)\n", sw);
244 }
245 else
246 {
247 /* We got a success response */
248 printf("Card UID=");
249 for (DWORD i = 0; i < dwRecvLength - 2; i++)
250 printf("%02X", abInOutBuffer[i]);
251 printf("\n");
252 }
253 }
254
255 if (fTestEchoTransmit)
256 {
257 printf("Please wait during the test (echo over SCardTransmit)...\n");
258 if (!test_echo_transmit(slot))
259 {
260 printf("Test failed!\n");
261 return;
262 }
263 }
264
265 /* Done with the card */
266 SCARD_LIB(Disconnect)(slot);
267}

269static WORD get_status_word(BYTE abRecvApdu[], DWORD dwRecvLength)
270{
271 if (abRecvApdu == NULL)
272 return 0xFFFF;
273 if (dwRecvLength < 2)
274 return 0xFFFF;
275 return (abRecvApdu[dwRecvLength - 2] * 0x0100) + abRecvApdu[dwRecvLength - 1];
276}

327static void dump_device_descriptor(void)
328{
329 DWORD dwRecvLength = sizeof(abInOutBuffer);
330 LONG rc = CCID_LIB(GetDescriptor)(1,0, abInOutBuffer, &dwRecvLength);
331 if (rc != SCARD_ERR(S_SUCCESS))
332 {
333 printf("Failed to read device's main descriptor (rc=%lX)\n", rc);
334 return;
335 }
336
337 if (dwRecvLength < 18)
338 {
339 printf("Wrong size for device descriptor (expected 18 bytes, got %lu)\n", dwRecvLength);
340 return;
341 }
342
343 if ((abInOutBuffer[0] != 18) || (abInOutBuffer[1] != 1))
344 {
345 printf("Wrong format for device descriptor (expected header is %02X%02X, got %02X%02X)\n", 18, 1, abInOutBuffer[0], abInOutBuffer[1]);
346 return;
347 }
348
349 printf("Device descriptor:\n");
350 printf("\tbLength: %d\n", abInOutBuffer[0]);
351 printf("\tbDescriptorType: %d\n", abInOutBuffer[1]);
352 printf("\tbcdUSB: %X.%02X\n", abInOutBuffer[3], abInOutBuffer[2]);
353 printf("\tbDeviceClass: %02X\n", abInOutBuffer[4]);
354 printf("\tbDeviceSubClass: %02X\n", abInOutBuffer[5]);
355 printf("\tbDeviceProtocol: %02X\n", abInOutBuffer[6]);
356 printf("\tbMaxPacketSize0: %d\n", abInOutBuffer[7]);
357 printf("\tidVendor: %02X%02X\n", abInOutBuffer[9], abInOutBuffer[8]);
358 printf("\tidProduct: %02X%02X\n", abInOutBuffer[11], abInOutBuffer[10]);
359 printf("\tbcdDevice: %X.%02X\n", abInOutBuffer[13], abInOutBuffer[12]);
360 printf("\tiManufacturer: %d\n", abInOutBuffer[14]);
361 printf("\tiProduct: %d\n", abInOutBuffer[15]);
362 printf("\tiSerialNumber: %d\n", abInOutBuffer[16]);
363 printf("\tbNumConfigurations: %d\n", abInOutBuffer[17]);
364 printf("\n");
365}

367static void dump_configuration_descriptor(void)
368{
369 DWORD dwRecvLength = sizeof(abInOutBuffer);
370 WORD wTotalLength;
371 LONG rc = CCID_LIB(GetDescriptor)(2, 0, abInOutBuffer, &dwRecvLength);
372 if (rc != SCARD_ERR(S_SUCCESS))
373 {
374 printf("Failed to read device's configuration descriptor (rc=%lX)\n", rc);
375 return;
376 }
377
378 if (dwRecvLength < 9)
379 {
380 printf("Wrong size for configuration descriptor (expected at least 9 bytes, got %lu)\n", dwRecvLength);
381 return;
382 }
383
384 if ((abInOutBuffer[0] != 9) || (abInOutBuffer[1] != 2))
385 {
386 printf("Wrong format for configuration descriptor (expected header is %02X%02X, got %02X%02X)\n", 9, 2, abInOutBuffer[0], abInOutBuffer[1]);
387 return;
388 }
389
390 wTotalLength = (abInOutBuffer[3] << 8) | abInOutBuffer[2];
391
392 printf("Configuration descriptor:\n");
393 printf("\tbLength: %d\n", abInOutBuffer[0]);
394 printf("\tbDescriptorType: %d\n", abInOutBuffer[1]);
395 printf("\twTotalLength: %d\n", wTotalLength);
396 printf("\tbNumInterfaces: %d\n", abInOutBuffer[4]);
397 printf("\tbConfigurationValue: %d\n", abInOutBuffer[5]);
398 printf("\tiConfiguration: %d\n", abInOutBuffer[6]);
399 printf("\tbmAttributes: %02X\n", abInOutBuffer[7]);
400 printf("\tbMaxPower: %d (2mA units, so %dmA)\n", abInOutBuffer[8], abInOutBuffer[8] * 2);
401 printf("\n");
402}

587static BOOL test_echo_control_ex(unsigned delay, unsigned Lc, unsigned Le)
588{
589 LONG rc;
590 DWORD dwSendLength = 0;
591 DWORD dwRecvLength;
592
593 printf("Control Test with Lc=%d, Le=%d, delay=%d...\n", Lc, Le, delay);
594
595 init_random((Lc << 8) | Le);
596
597 abInOutBuffer[dwSendLength++] = 0xFF; /* CLA */
598 abInOutBuffer[dwSendLength++] = 0xFD; /* INS */
599 abInOutBuffer[dwSendLength++] = 0x00; /* P1 */
600 abInOutBuffer[dwSendLength++] = 0x80 | (delay & 0x3F); /* P2 */
601
602 if ((Lc == 0) && (Le == 0))
603 {
604 /* Case 1 APDU */
605 }
606 else if ((Lc == 0) && (Le > 0))
607 {
608 /* Case 2 APDU */
609 abInOutBuffer[dwSendLength++] = (BYTE) Le; /* 256 --> 0 is correct */
610 }
611 else if ((Lc > 0) && (Le == 0))
612 {
613 /* Case 3 APDU */
614 abInOutBuffer[dwSendLength++] = (BYTE)Lc;
615 /* Add some "random" data */
616 for (unsigned i = 0; i < Lc; i++)
617 abInOutBuffer[dwSendLength++] = get_random();
618 }
619 else /* ((Lc > 0) && (Le > 0)) */
620 {
621 /* Case 4 APDU */
622 abInOutBuffer[dwSendLength++] = (BYTE)Lc;
623 /* Add some "random" data */
624 for (unsigned i = 0; i < Lc; i++)
625 abInOutBuffer[dwSendLength++] = get_random();
626 /* And terminate with Le */
627 abInOutBuffer[dwSendLength++] = (BYTE)Le; /* 256 --> 0 is correct */
628 }
629
630 /* Transmit */
631 dwRecvLength = sizeof(abInOutBuffer);
632 rc = SCARD_LIB(Control)(abInOutBuffer, dwSendLength, abInOutBuffer, &dwRecvLength);
633 if (rc != SCARD_ERR(S_SUCCESS))
634 {
635 printf("Test failed for Lc=%d, Le=%d, delay=%d (rc=%lX)\n", Lc, Le, delay, rc);
636 return FALSE;
637 }
638 else
639 {
640 WORD sw = get_status_word(abInOutBuffer, dwRecvLength);
641 if (sw != 0x9000)
642 {
643 printf("Test returned SW=%04X for Lc=%d, Le=%d\n", sw, Lc, Le);
644 return FALSE;
645 }
646 if (dwRecvLength != (Le + 2))
647 {
648 printf("Test returned %lu bytes instead of %u for Lc=%u, Le=%u\n", dwRecvLength, Le + 2, Lc, Le);
649 return FALSE;
650 }
651
652 if (Lc == Le)
653 {
654 /* Test that echo is correct */
655 init_random((Lc << 8) | Le);
656
657 for (unsigned i = 0; i < Lc; i++)
658 {
659 if (abInOutBuffer[i] != get_random())
660 {
661 printf("Test failed, returned data is not an echo of sent data (at offset %d) for Lc=%d, Le=%d\n", i, Lc, Le);

700static BOOL test_echo_transmit_ex(BYTE slot, unsigned delay, unsigned Lc, unsigned Le)
701{
702 LONG rc;
703 DWORD dwSendLength = 0;
704 DWORD dwRecvLength;
705
706 printf("Transmit Test with Lc=%d, Le=%d, delay=%d...\n", Lc, Le, delay);
707
708 init_random((Lc << 8) | Le);
709
710 abInOutBuffer[dwSendLength++] = 0xFF; /* CLA */
711 abInOutBuffer[dwSendLength++] = 0xFD; /* INS */
712 abInOutBuffer[dwSendLength++] = 0x00; /* P1 */
713 abInOutBuffer[dwSendLength++] = 0x80 | (delay & 0x3F); /* P2 */
714
715 if ((Lc == 0) && (Le == 0))
716 {
717 /* Case 1 APDU */
718 }
719 else if ((Lc == 0) && (Le > 0))
720 {
721 /* Case 2 APDU */
722 abInOutBuffer[dwSendLength++] = (BYTE) Le; /* 256 --> 0 is correct */
723 }
724 else if ((Lc > 0) && (Le == 0))
725 {
726 /* Case 3 APDU */
727 abInOutBuffer[dwSendLength++] = (BYTE)Lc;
728 /* Add some "random" data */
729 for (unsigned i = 0; i < Lc; i++)
730 abInOutBuffer[dwSendLength++] = get_random();
731 }
732 else /* ((Lc > 0) && (Le > 0)) */
733 {
734 /* Case 4 APDU */
735 abInOutBuffer[dwSendLength++] = (BYTE)Lc;
736 /* Add some "random" data */
737 for (unsigned i = 0; i < Lc; i++)
738 abInOutBuffer[dwSendLength++] = get_random();
739 /* And terminate with Le */
740 abInOutBuffer[dwSendLength++] = (BYTE)Le; /* 256 --> 0 is correct */
741 }
742
743 /* Transmit */
744 dwRecvLength = sizeof(abInOutBuffer);
745 rc = SCARD_LIB(Transmit)(slot, abInOutBuffer, dwSendLength, abInOutBuffer, &dwRecvLength);
746 if (rc != SCARD_ERR(S_SUCCESS))
747 {
748 printf("Test failed for Lc=%d, Le=%d, delay=%d (rc=%lX)\n", Lc, Le, delay, rc);
749 return FALSE;
750 }
751 else
752 {
753 WORD sw = get_status_word(abInOutBuffer, dwRecvLength);
754 if (sw != 0x9000)
755 {
756 printf("Test returned SW=%04X for Lc=%d, Le=%d\n", sw, Lc, Le);
757 return FALSE;
758 }
759 if (dwRecvLength != (Le + 2))
760 {
761 printf("Test returned %lu bytes instead of %u for Lc=%u, Le=%u\n", dwRecvLength, Le + 2, Lc, Le);
762 return FALSE;
763 }
764
765 if (Lc == Le)
766 {
767 /* Test that echo is correct */
768 init_random((Lc << 8) | Le);
769
770 for (unsigned i = 0; i < Lc; i++)
771 {
772 if (abInOutBuffer[i] != get_random())
773 {
774 printf("Test failed, returned data is not an echo of sent data (at offset %d) for Lc=%d, Le=%d\n", i, Lc, Le);
775 return FALSE;
776 }
777 }
778 }
779 }
780
781 return TRUE;
782}

784static BOOL test_echo_transmit(BYTE slot)
785{
786 /* Test the limits first - It is better to fail quickly if something is wrong */
787 if (!test_echo_transmit_ex(slot, 0, 0, 0))
788 return FALSE;
789 if (!test_echo_transmit_ex(slot, 1, 0, 0))

799 if (!test_echo_transmit_ex(slot, 60, 0, 0))
800 return FALSE;

803 for (int Lc = 1; Lc <= 255; Lc++)

813 }
814 /* And test actual echo (recv length = send length) */
815 for (int Lc = 1; Lc <= 255; Lc++)

817 if (!test_echo_transmit_ex(slot, 0, Lc, Lc))
818 return FALSE;
819 }
