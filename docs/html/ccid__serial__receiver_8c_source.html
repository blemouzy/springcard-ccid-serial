109void CCID_LIB(SerialRecvByteFromISR)(BYTE bValue)
110{
111 CCID_RECEIVER_ST* receiver;
112
113 if (ccid_receiver_error)
114 return; /* Stop receiving until the error is cleared */
115
116 /* Make sure we have a valid receiver */
117 ccid_receiver_push_index %= 2;
118 receiver = &ccid_receivers[ccid_receiver_push_index];
119
120 switch (receiver->bStatus)
121 {
122 case STATUS_IDLE:
123 if (bValue == START_BYTE)
124 {
125 /* This is the beginning of a serial CCID message */
126 memset(receiver, 0, sizeof(CCID_RECEIVER_ST));
127 receiver->bStatus = STATUS_RECV_ENDPOINT;
128 }
129 else
130 {
131 /* Invalid byte */
132 ccid_receiver_error = TRUE;
133 receiver->bStatus = STATUS_ERROR_PROTOCOL;
134 CCID_LIB(WakeupFromISR)();
135 }
136 break;
137
138 case STATUS_RECV_ENDPOINT:
139 /* Init the data */
140 receiver->bEndpoint = bValue;
141 receiver->bChecksum = bValue;
142 /* Now ready to receive the header */
143 receiver->dwLength = CCID_HEADER_LENGTH;
144 receiver->dwOffset = 0;
145 receiver->bStatus = STATUS_RECV_HEADER;
146 break;
147
148 case STATUS_RECV_HEADER:
149 receiver->bChecksum ^= bValue;
150 receiver->abBuffer[receiver->dwOffset++] = bValue;
151 if (receiver->dwOffset >= CCID_HEADER_LENGTH)
152 {
153 DWORD dwLength = utohl(&receiver->abBuffer[CCID_POS_LENGTH]);
154
155 if (dwLength > CCID_MAX_PAYLOAD_LENGTH)
156 {
157 /* Payload will not fit in our buffer */
158 ccid_receiver_error = TRUE;
159 receiver->bStatus = STATUS_ERROR_OVERFLOW;
160 CCID_LIB(WakeupFromISR)();
161 }
162 else if (dwLength)
163 {
164 /* Ready to receive the payload */
165 receiver->dwLength = CCID_HEADER_LENGTH + dwLength;
166 receiver->bStatus = STATUS_RECV_PAYLOAD;
167 }
168 else
169 {
170 /* No payload, ready to receive the checksum */
171 receiver->bStatus = STATUS_RECV_CHECKSUM;
172 }
173 }
174 break;
175
176 case STATUS_RECV_PAYLOAD:
177 receiver->bChecksum ^= bValue;
178 receiver->abBuffer[receiver->dwOffset++] = bValue;
179 if (receiver->dwOffset >= receiver->dwLength)
180 {
181 /* Done with the payload, ready to receive the checksum */
182 receiver->bStatus = STATUS_RECV_CHECKSUM;
183 }
184 break;
185
186 case STATUS_RECV_CHECKSUM:
187 receiver->bChecksum ^= bValue;
188 if (!receiver->bChecksum)
189 {
190 /* Checkum is OK */
191 receiver->bStatus = STATUS_READY;
192 /* Toggle */
193 ccid_receiver_push_index = 1 - ccid_receiver_push_index;
194 /* Wakeup the application */
195 CCID_LIB(WakeupFromISR)();
196 }
197 else
198 {
199 ccid_receiver_error = TRUE;
200 receiver->bStatus = STATUS_ERROR_CHECKSUM;
201 CCID_LIB(WakeupFromISR)();
202 }
203 break;
204
205 case STATUS_READY:
206 ccid_receiver_error = TRUE;
207 dump();
208 printf("ERROR OVERRUN !!! %02X %d\n", bValue, receiver->bStatus);
209 receiver->bStatus = STATUS_ERROR_OVERRUN;
210 CCID_LIB(WakeupFromISR)();
211 break;
212
213 default:
214 receiver->bStatus = STATUS_ERROR_UNEXPECTED;
215 CCID_LIB(WakeupFromISR)();
216 }
217}

223LONG CCID_LIB(SerialRecv)(CCID_PACKET_ST* packet, DWORD timeout_ms)
224{
225 LONG rc = SCARD_ERR(S_SUCCESS);
226 CCID_RECEIVER_ST* receiver;
227
228 if (packet == NULL)
229 return SCARD_ERR(E_INVALID_PARAMETER);
230
231 ccid_receiver_pop_index %= 2;
232 receiver = &ccid_receivers[ccid_receiver_pop_index];
233
234 CCID_LIB(ClearWakeup)();
235
236 if (receiver->bStatus != STATUS_READY)
237 {
238 /* Wait until a message arrives */
239 if (!CCID_LIB(WaitWakeup)(timeout_ms))
240 rc = SCARD_ERR(E_TIMEOUT);
241 }
242
243 if (rc == SCARD_ERR(S_SUCCESS))
244 {
245 switch (receiver->bStatus)
246 {
247 case STATUS_READY:
248 /* This is the expected situation */
249 break;
250 case STATUS_IDLE:
251 rc = SCARD_ERR(E_NOT_READY); /* Nothing received */
252 break;
253 case STATUS_ERROR_PROTOCOL:
254 rc = SCARD_ERR(E_READER_UNSUPPORTED); /* Wrong protocol */
255 break;
256 case STATUS_ERROR_CHECKSUM:
257 rc = SCARD_ERR(F_COMM_ERROR); /* Wrong checksum */
258 break;
259 case STATUS_ERROR_OVERFLOW:
260 rc = SCARD_ERR(E_NO_MEMORY); /* Internal buffer is too short */
261 break;
262 case STATUS_ERROR_OVERRUN:
263 rc = SCARD_ERR(F_INTERNAL_ERROR); /* Sequence was incoherent */
264 break;
265 default:
266 rc = SCARD_ERR(E_UNEXPECTED); /* Oops? Whats going on */
267 }
268 }
269
270 if (ccid_receiver_error)
271 {
272 /* Make sure the application knows there is an error */
273 if (rc == SCARD_ERR(S_SUCCESS))
274 rc = SCARD_ERR(F_UNKNOWN_ERROR);
275
276 /* Cleanup */
277 memset(ccid_receivers, 0, sizeof(ccid_receivers));
278 ccid_receiver_push_index = 0;
279 ccid_receiver_pop_index = 0;
280 ccid_receiver_error = FALSE;
281 }
282
283 if (rc == SCARD_ERR(S_SUCCESS))
284 {
285 packet->bEndpoint = receiver->bEndpoint;
286 memcpy(packet->Header.u, receiver->abBuffer, CCID_HEADER_LENGTH);
287 packet->Header.p.Length.dw = utohl(&receiver->abBuffer[CCID_POS_LENGTH]);
288
289 if (packet->bEndpoint == CCID_COMM_CONTROL_TO_PC)
290 {
291 packet->Header.p.Data.Control.Value.w = utohs(packet->Header.p.Data.Control.Value.ab);
292 packet->Header.p.Data.Control.Index.w = utohs(packet->Header.p.Data.Control.Index.ab);
293 }
294
295 if (packet->Header.p.Length.dw)
296 {
297 if ((packet->abRecvPayload == NULL) || (packet->dwRecvPayloadMaxLen < packet->Header.p.Length.dw))
298 {
299 rc = SCARD_ERR(E_INSUFFICIENT_BUFFER);
300 }
301 else
302 {
303 memcpy(packet->abRecvPayload, &receiver->abBuffer[CCID_HEADER_LENGTH], packet->Header.p.Length.dw);
304 }
305 }
306 }
307 else
308 {
309 dump();
310 }
311
312 /* This receiver is ready to receive again */
313 receiver->bStatus = STATUS_IDLE;
314 /* And next time we'll read the other */
315 ccid_receiver_pop_index = 1 - ccid_receiver_pop_index;
316
317 D(printf("\n"));
318 return rc;
319}
