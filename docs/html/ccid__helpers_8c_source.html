122LONG CCID_LIB(Start)(BOOL fUseNotifications)
123{
124 CCID_PACKET_ST packet;
125 LONG rc;
126
127 ccid_clear_error();
128
129 CCID_LIB(PacketInit)(&packet);
130
131 packet.bEndpoint = CCID_COMM_CONTROL_TO_RDR;
132 packet.Header.p.bRequest = SET_CONFIGURATION;
133 packet.Header.p.Data.Control.Value.w = 1;
134 packet.Header.p.Data.Control.Index.w = 0;
135 packet.Header.p.Data.Control.InOut.bOutOption = 0;
136
137 if (fUseNotifications)
138 {
139 packet.Header.p.Data.Control.InOut.bOutOption = 1;
140 }
141
142 rc = CCID_LIB(Exchange)(&packet, CONTROL_TIMEOUT);
143
144 if (rc == SCARD_ERR(S_SUCCESS))
145 {
146 if (packet.Header.p.Data.Control.InOut.bInStatus != 0x01)
147 rc = SCARD_ERR(E_UNEXPECTED);
148 }
149
150 /* Reset the sequence numbers */
151 CCID_LIB(ResetSequences)();
152
153 return rc;
154}

188LONG CCID_LIB(GetDescriptor)(BYTE bType, BYTE bIndex, BYTE abDescriptor[], DWORD *pdwDescriptorLength)
189{
190 CCID_PACKET_ST packet;
191 LONG rc;
192
193 if ((abDescriptor != NULL) && (pdwDescriptorLength == NULL))
194 return SCARD_ERR(E_INVALID_PARAMETER);
195
196 CCID_LIB(PacketInit)(&packet);
197
198 packet.bEndpoint = CCID_COMM_CONTROL_TO_RDR;
199 packet.Header.p.bRequest = GET_DESCRIPTOR;
200 packet.Header.p.Data.Control.Value.ab[0] = bType;
201 packet.Header.p.Data.Control.Value.ab[1] = bIndex;
202
203 if (pdwDescriptorLength != NULL)
204 {
205 packet.abRecvPayload = abDescriptor;
206 packet.dwRecvPayloadMaxLen = *pdwDescriptorLength;
207 }
208
209 rc = CCID_LIB(Exchange)(&packet, CONTROL_TIMEOUT);
210
211 if (rc == SCARD_ERR(S_SUCCESS))
212 {
213 if (packet.Header.p.Data.Control.InOut.bInStatus != 0x00)
214 {
215 rc = SCARD_ERR(E_UNEXPECTED);
216 }
217 else
218 {
219 if (pdwDescriptorLength != NULL)
220 *pdwDescriptorLength = packet.Header.p.Length.dw;
221 }
222 }
223
224 return rc;
225}

230LONG CCID_LIB(GetSlotCount)(BYTE* bSlotCount)
231{
232 /*
233 * We use the upper layer (SCARD aka PC/SC) to answer a query regarding the low layer...
234 * I know this is not OSI compliant ;-)
235 * Why doing so?
236 * Just because the device is able to return the slot count at once, where reading and
237 * parsing its descriptor requires more memory and more communication time.
238 * */
239 const BYTE abSendBuffer[] = { 0x58, 0x20, 0x80 };
240 BYTE abRecvBuffer[2];
241 DWORD dwRecvLength = sizeof(abRecvBuffer);
242 LONG rc = SCARD_LIB(Control)(abSendBuffer, sizeof(abSendBuffer), abRecvBuffer, &dwRecvLength);
243
244 if (rc == SCARD_ERR(S_SUCCESS))
245 {
246 if (dwRecvLength < 2)
247 {
248 rc = SCARD_ERR(E_READER_UNSUPPORTED);
249 }
250 else if (abRecvBuffer[0] != 0x00)
251 {
252 rc = SCARD_ERR(E_READER_UNSUPPORTED);
253 }
254 else
255 {
256 if (bSlotCount != NULL)
257 *bSlotCount = abRecvBuffer[1];
258 }
259 }
260
261 return rc;
262}
