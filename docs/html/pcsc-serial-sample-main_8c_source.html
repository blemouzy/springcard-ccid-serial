77int main(int argc, char** argv)
78{
79 LONG rc;
80
81 printf("\n");
82 printf("SpringCard SDK for PC/SC Serial : demo for Windows and Linux\n");
83 printf("------------------------------------------------------------\n");
84 printf("Copyright (c) 2015-2023 SPRINGCARD SAS, FRANCE - www.springcard.com\n");
85 printf("See LICENSE.txt for disclaimer and license requirements\n");
86 printf("\n");
87
88 if (!parse_args(argc, argv))
89 {
90 printf("Usage:\n");
91 printf("\tpcsc-serial-windows-demo [-d <COMM PORT>] [-i] [-c] [-t] [-v]\n");
92 printf("\t\t-d <COM PORT>: select the comm. device (default is COM5)\n");
93 printf("\t\t-i: use notifications (Interrupt endpoint)\n");
94 printf("\t\t-c: run ECHO test over SCardControl\n");
95 printf("\t\t-t: run ECHO test over SCardTransmit\n");
96 printf("\t\t-v: verbose output\n");
97 return -1;
98 }
99
100 printf("Using communication device: %s\n", szCommDevice);
101
102 /* Prepare the underlying hardware and lower layer software */
103 /* -------------------------------------------------------- */
104
105 CCID_LIB(SerialInit)(szCommDevice);
106
107 printf("\n\n***** Now running forever, hit Ctrl+C to exit ****\n\n");
108
109 for (;;)
110 {
111 /* Try and open the serial communication layer */
112 /* ------------------------------------------- */
113 printf("Opening the serial port\n");
114 if (!CCID_LIB(SerialOpen)())
115 {
116 printf("Failed to open serial port %s\n", szCommDevice);
117 sleep_ms(1000);
118 continue;
119 }
120
121 /* Initialize the CCID driver */
122 /* -------------------------- */
123
124 printf("Initializing the CCID driver\n");
125 CCID_LIB(Init)();
126
127 /* Connect or reconnect to the device */
128 /* ---------------------------------- */
129
130 printf("Pinging the device (if some)\n");
131 rc = CCID_LIB(Ping)();
132 if (rc != SCARD_ERR(S_SUCCESS))
133 {
134 printf("No device found on port %s (rc=%lX)\n", szCommDevice, rc);
135 /* Close the serial port */
136 CCID_LIB(SerialClose)();
137 /* In case of a communication error, we shall wait at least 1200ms so the device may reset its state machine */
138 sleep_ms(1200);
139 /* Try again */
140 continue;
141 }
142
143 printf("*** It seems that we've found a device... ***\n");
144
145 /* Initialize the PC/SC-like stack */
146 /* ------------------------------- */
147
148 printf("Initializing the PC/SC-Like stack\n");
149 SCARD_LIB(Init)();
150
151 /* Run the sample */
152 /* --------------- */
153
154 sample();
155
156 /* The sample is a loop, if we come back here, this means that we have lost the device */
157 /* ----------------------------------------------------------------------------------- */
158 printf("*** It seems that we've lost the device... ***\n");
159
160 /* Close the serial port */
161 printf("Closing the serial port\n");
162 CCID_LIB(SerialClose)();
163 }
164
165 return 0;
166}

169static BOOL parse_args(int argc, char** argv)
170{
171 for (int i = 1; i < argc; i++)
172 { // Start at 1 because argv[0] is the program name
173 if (argv[i][0] == '-')
174 { // This is an option
175 if (!strcmp(argv[i], "-d") && i + 1 < argc)
176 {
177 szCommDevice = argv[i + 1];
178 i++; // Skip next item since we just processed it
179 }
180 else if (!strcmp(argv[i], "-i"))
181 {
182 fCcidUseNotifications = TRUE;
183 }
184 else if (!strcmp(argv[i], "-c"))
185 {
186 fTestEchoControl = TRUE;
187 }
188 else if (!strcmp(argv[i], "-t"))
189 {
190 fTestEchoTransmit = TRUE;
191 }
192 else if (!strcmp(argv[i], "-v"))
193 {
194 fVerbose = TRUE;
195 }
196 else if (!strcmp(argv[i], "-h"))
197 {
198 /* Return FALSE to display the usage message */
199 return FALSE;
200 }
201 else
202 {
203 printf("Unknown option: %s\n", argv[i]);
204 return FALSE;
205 }
206 }
207 else
208 {
209 // It's not an option, maybe a standalone argument
210 printf("Unsupported argument: %s\n", argv[i]);
211 return FALSE;
212 }
213 }
214
215 return TRUE;
216}
