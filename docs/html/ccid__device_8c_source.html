68LONG CCID_LIB(Start)(BOOL fUseNotifications)
69{
70 CCID_PACKET_ST packet;
71 LONG rc;
72
73 ccid_clear_error();
74
75 CCID_LIB(PacketInit)(&packet);
76
77 packet.bEndpoint = CCID_COMM_CONTROL_TO_RDR;
78 packet.Header.p.bRequest = SET_CONFIGURATION;
79 packet.Header.p.Data.Control.Value.w = 1;
80 packet.Header.p.Data.Control.Index.w = 0;
81 packet.Header.p.Data.Control.InOut.bOutOption = 0;
82
83 if (fUseNotifications)
84 {
85 packet.Header.p.Data.Control.InOut.bOutOption = 1;
86 }
87
88 rc = CCID_LIB(Exchange)(&packet, CONTROL_TIMEOUT);
89
90 if (rc == SCARD_ERR(S_SUCCESS))
91 {
92 if (packet.Header.p.Data.Control.InOut.bInStatus != 0x01)
93 rc = SCARD_ERR(E_UNEXPECTED);
94 }
95
96 /* Reset the sequence numbers */
97 CCID_LIB(ResetSequences)();
98
99 return rc;
100}

136LONG CCID_LIB(GetDescriptor)(BYTE bType, BYTE bIndex, BYTE abDescriptor[], DWORD *pdwDescriptorLength)
137{
138 CCID_PACKET_ST packet;
139 LONG rc;
140
141 if ((abDescriptor != NULL) && (pdwDescriptorLength == NULL))
142 return SCARD_ERR(E_INVALID_PARAMETER);
143
144 CCID_LIB(PacketInit)(&packet);
145
146 packet.bEndpoint = CCID_COMM_CONTROL_TO_RDR;
147 packet.Header.p.bRequest = GET_DESCRIPTOR;
148 packet.Header.p.Data.Control.Value.ab[0] = bType;
149 packet.Header.p.Data.Control.Value.ab[1] = bIndex;
150
151 if (pdwDescriptorLength != NULL)
152 {
153 packet.abRecvPayload = abDescriptor;
154 packet.dwRecvPayloadMaxLen = *pdwDescriptorLength;
155 }
156
157 rc = CCID_LIB(Exchange)(&packet, CONTROL_TIMEOUT);
158
159 if (rc == SCARD_ERR(S_SUCCESS))
160 {
161 if (packet.Header.p.Data.Control.InOut.bInStatus != 0x00)
162 {
163 rc = SCARD_ERR(E_UNEXPECTED);
164 }
165 else
166 {
167 if (pdwDescriptorLength != NULL)
168 *pdwDescriptorLength = packet.Header.p.Length.dw;
169 }
170 }
171
172 return rc;
173}

175LONG CCID_LIB(GetSlotCount)(BYTE* bSlotCount)
176{
177 /*
178 * We use the upper layer (SCARD aka PC/SC) to answer a query regarding the low layer...
179 * I know this is not OSI compliant ;-)
180 * Why doing so?
181 * Just because the device is able to return the slot count at once, where reading and
182 * parsing its descriptor requires more memory and more communication time.
183 * */
184 const BYTE abSendBuffer[] = { 0x58, 0x20, 0x80 };
185 BYTE abRecvBuffer[2];
186 DWORD dwRecvLength = sizeof(abRecvBuffer);
187 LONG rc = SCARD_LIB(Control)(abSendBuffer, sizeof(abSendBuffer), abRecvBuffer, &dwRecvLength);
188
189 if (rc == SCARD_ERR(S_SUCCESS))
190 {
191 if (dwRecvLength < 2)
192 {
193 rc = SCARD_ERR(E_READER_UNSUPPORTED);
194 }
195 else if (abRecvBuffer[0] != 0x00)
196 {
197 rc = SCARD_ERR(E_READER_UNSUPPORTED);
198 }
199 else
200 {
201 if (bSlotCount != NULL)
202 *bSlotCount = abRecvBuffer[1];
203 }
204 }
205
206 return rc;
207}
