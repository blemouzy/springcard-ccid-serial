58LONG CCID_LIB(SerialSend)(CCID_PACKET_ST* packet)
59{
60 DWORD i, dwSendPayloadLength;
61 BYTE bChecksum;
62
63 if (packet == NULL)
64 return SCARD_ERR(E_INVALID_PARAMETER);
65
66 dwSendPayloadLength = packet->Header.p.Length.dw;
67 htoul(packet->Header.p.Length.ab, dwSendPayloadLength);
68
69 if (packet->bEndpoint == CCID_COMM_CONTROL_TO_RDR)
70 {
71 htous(packet->Header.p.Data.Control.Value.ab, packet->Header.p.Data.Control.Value.w);
72 htous(packet->Header.p.Data.Control.Index.ab, packet->Header.p.Data.Control.Index.w);
73 }
74
75 if ((dwSendPayloadLength != 0) && (packet->abSendPayload == NULL))
76 return SCARD_ERR(E_INVALID_PARAMETER);
77
78 bChecksum = packet->bEndpoint;
79 for (i = 0; i < CCID_HEADER_LENGTH; i++)
80 bChecksum ^= packet->Header.u[i];
81
82 if (packet->abSendPayload != NULL)
83 for (i = 0; i < dwSendPayloadLength; i++)
84 bChecksum ^= packet->abSendPayload[i];
85
86 if (!CCID_LIB(SerialSendByte)(START_BYTE))
87 return SCARD_ERR(F_COMM_ERROR);
88 if (!CCID_LIB(SerialSendByte)(packet->bEndpoint))
89 return SCARD_ERR(F_COMM_ERROR);
90 if (!CCID_LIB(SerialSendBytes)(packet->Header.u, CCID_HEADER_LENGTH))
91 return SCARD_ERR(F_COMM_ERROR);
92 if (packet->abSendPayload != NULL)
93 if (!CCID_LIB(SerialSendBytes)(packet->abSendPayload, dwSendPayloadLength))
94 return SCARD_ERR(F_COMM_ERROR);
95 if (!CCID_LIB(SerialSendByte)(bChecksum))
96 return SCARD_ERR(F_COMM_ERROR);
97
98 return SCARD_ERR(S_SUCCESS);
99}
