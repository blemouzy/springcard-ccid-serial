44LONG SCARD_LIB(GetStatusChangeEx)(DWORD dwTimeoutMs, DWORD* pdwPresentSlots, DWORD* pdwChangedSlots)
45{
46 CCID_PACKET_ST packet;
47 BYTE abInterruptBuffer[CCID_MAX_INTERRUPT_PAYLOAD_LENGTH];
48 LONG rc;
49 DWORD dwPresentSlots = 0;
50 DWORD dwChangedSlots = 0;
51
52 CCID_LIB(PacketInit)(&packet);
53
54 packet.abRecvPayload = abInterruptBuffer;
55 packet.dwRecvPayloadMaxLen = sizeof(abInterruptBuffer);
56
57 rc = CCID_LIB(WaitInterrupt)(&packet, dwTimeoutMs);
58
59 if (rc == SCARD_ERR(S_SUCCESS))
60 {
61 if (packet.Header.p.Length.dw > 1)
62 {
63 if (abInterruptBuffer[0] & 0x01)
64 dwPresentSlots |= 0x00000001;
65 if (abInterruptBuffer[0] & 0x02)
66 dwChangedSlots |= 0x00000001;
67 if (abInterruptBuffer[0] & 0x04)
68 dwPresentSlots |= 0x00000002;
69 if (abInterruptBuffer[0] & 0x08)
70 dwChangedSlots |= 0x00000002;
71 if (abInterruptBuffer[0] & 0x10)
72 dwPresentSlots |= 0x00000004;
73 if (abInterruptBuffer[0] & 0x20)
74 dwChangedSlots |= 0x00000004;
75 if (abInterruptBuffer[0] & 0x40)
76 dwPresentSlots |= 0x00000008;
77 if (abInterruptBuffer[0] & 0x80)
78 dwChangedSlots |= 0x00000008;
79 }
80 if (packet.Header.p.Length.dw > 2)
81 {
82 if (abInterruptBuffer[1] & 0x01)
83 dwPresentSlots |= 0x00000010;
84 if (abInterruptBuffer[1] & 0x02)
85 dwChangedSlots |= 0x00000010;
86 if (abInterruptBuffer[1] & 0x04)
87 dwPresentSlots |= 0x00000020;
88 if (abInterruptBuffer[1] & 0x08)
89 dwChangedSlots |= 0x00000020;
90 if (abInterruptBuffer[1] & 0x10)
91 dwPresentSlots |= 0x00000040;
92 if (abInterruptBuffer[1] & 0x20)
93 dwChangedSlots |= 0x00000040;
94 if (abInterruptBuffer[1] & 0x40)
95 dwPresentSlots |= 0x00000080;
96 if (abInterruptBuffer[1] & 0x80)
97 dwChangedSlots |= 0x00000080;
98 }
99
100 D(printf("Interrupt, slots present: %08lX, slots changed: %08lX\n", dwPresentSlots, dwChangedSlots));
101
102 if (pdwPresentSlots != NULL)
103 *pdwPresentSlots = dwPresentSlots;
104 if (pdwChangedSlots != NULL)
105 *pdwChangedSlots = dwChangedSlots;
106 }
107
108 return rc;
109}

111LONG SCARD_LIB(GetStatusChange)(DWORD dwTimeoutMs)
112{
113 return SCARD_LIB(GetStatusChangeEx)(dwTimeoutMs, NULL, NULL);
114}
