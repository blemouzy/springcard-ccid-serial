105static LONG ccid_recv_to_slot_error(BYTE bSlotError)
106{
107 LONG rc;
108
109 switch (bSlotError)
110 {
111 case CCID_ERR_UNKNOWN:
112 case CCID_ERR_PARAMETERS:
113 case CCID_ERR_CMD_NOT_SUPPORTED:
114 case CCID_ERR_BAD_LENGTH:
115 case CCID_ERR_BAD_SLOT:
116 case CCID_ERR_BAD_POWERSELECT:
117 case CCID_ERR_BAD_LEVELPARAMETER:
118 case CCID_ERR_BAD_FIDI:
119 case CCID_ERR_BAD_T01CONVCHECKSUM:
120 case CCID_ERR_BAD_GUARDTIME:
121 case CCID_ERR_BAD_WAITINGINTEGER:
122 case CCID_ERR_BAD_CLOCKSTOP:
123 case CCID_ERR_BAD_IFSC:
124 case CCID_ERR_BAD_NAD:
125 case CCID_ERR_CMD_ABORTED:
126 rc = SCARD_ERR(E_UNEXPECTED);
127 break;
128
129 case CCID_ERR_ICC_MUTE:
130 case CCID_ERR_XFR_PARITY_ERROR:
131 case CCID_ERR_XFR_OVERRUN:
132 case CCID_ERR_HW_ERROR:
133 rc = SCARD_ERR(W_UNRESPONSIVE_CARD);
134 break;
135
136 case CCID_ERR_BAD_ATR_TS:
137 case CCID_ERR_BAD_ATR_TCK:
138 case CCID_ERR_ICC_PROTOCOL_NOT_SUPPORTED:
139 case CCID_ERR_ICC_CLASS_NOT_SUPPORTED:
140 case CCID_ERR_PROCEDURE_BYTE_CONFLICT:
141 case CCID_ERR_DEACTIVATED_PROTOCOL:
142 rc = SCARD_ERR(W_UNSUPPORTED_CARD);
143 break;
144
145 case CCID_ERR_BUSY_WITH_AUTO_SEQUENCE:
146 case CCID_ERR_CMD_SLOT_BUSY:
147 rc = SCARD_ERR(E_UNEXPECTED);
148 break;
149
150 case CCID_SUCCESS:
151 default:
152 rc = SCARD_ERR(S_SUCCESS);
153 break;
154 }

156 return rc;

163static LONG ccid_recv_to_slot_status(CCID_PACKET_ST* packet)
164{
165 switch (packet->Header.p.Data.BulkIn.bSlotStatus & 0xC0)
166 {
167 case 0x00:
168 break;
169 case 0x40:
170 return ccid_recv_to_slot_error(packet->Header.p.Data.BulkIn.bSlotError);
171 case 0x80:
172 return SCARD_ERR(E_TIMEOUT);
173 case 0xC0:
174 default:
175 return SCARD_ERR(E_READER_UNSUPPORTED);
176 }
177
178 switch (packet->Header.p.Data.BulkIn.bSlotStatus & 0x03)
179 {
180 case 0x00:
181 return SCARD_ERR(S_SUCCESS);
182 case 0x01:
183 return SCARD_ERR(W_UNRESPONSIVE_CARD);
184 case 0x02:
185 return SCARD_ERR(W_REMOVED_CARD);
186 case 0x03:
187 default:
188 return SCARD_ERR(E_READER_UNSUPPORTED);
189 }
190}

195LONG CCID_LIB(Exchange)(CCID_PACKET_ST* packet, DWORD timeout_ms)
196{
197 LONG rc;
198 BYTE bEndpoint;
199 BYTE bSlot, bSequence;
200 WORD wIndex, wValue;
201 WORD wTimeExtension = 0;
202
203 if (packet == NULL)
204 {
205 ccid_raise_error("NULL packet in CCID_Exchange");
206 return SCARD_ERR(F_INTERNAL_ERROR);
207 }
208
209 bEndpoint = packet->bEndpoint;
210 wIndex = packet->Header.p.Data.Control.Index.w;
211 wValue = packet->Header.p.Data.Control.Value.w;
212 bSlot = packet->Header.p.Data.BulkOut.bSlot;
213 bSequence = packet->Header.p.Data.BulkOut.bSequence;
214
215 rc = CCID_LIB(SerialSend)(packet);
216 if (rc != SCARD_ERR(S_SUCCESS))
217 {
218 ccid_raise_error("Failed to send packet to device");
219 return rc;
220 }
221
222again:
223
224 rc = CCID_LIB(SerialRecv)(packet, timeout_ms);
225 if (rc != SCARD_ERR(S_SUCCESS))
226 {
227 ccid_raise_error("Failed to receive packet from device");
228 return rc;
229 }
230
231 if (packet->bEndpoint == CCID_COMM_INTERRUPT_RDR_TO_PC)
232 {
233 /* This is not a response but an interrupt. We can discard it safely if we are in the middle of an exchange */
234 D(printf("Incoming Interrupt\n"));
235 goto again;
236 }
237
238 switch (bEndpoint)
239 {
240 case CCID_COMM_CONTROL_TO_RDR:
241 if (packet->bEndpoint != CCID_COMM_CONTROL_TO_PC)
242 {
243 ccid_raise_error("Wrong endpoint in response");
244 rc = SCARD_ERR(E_READER_UNSUPPORTED);
245 }
246 else if ((packet->Header.p.Data.Control.Index.w != wIndex) || (packet->Header.p.Data.Control.Value.w != wValue))
247 {
248 ccid_raise_error("Wrong index/value in response");
249 rc = SCARD_ERR(E_READER_UNSUPPORTED);
250 }
251 break;
252
253 case CCID_COMM_BULK_PC_TO_RDR:
254 if (packet->bEndpoint != CCID_COMM_BULK_RDR_TO_PC)
255 {
256 ccid_raise_error("Wrong endpoint in response");
257 rc = SCARD_ERR(E_READER_UNSUPPORTED);
258 }
259 else if ((packet->Header.p.Data.BulkIn.bSlot != bSlot) || (packet->Header.p.Data.BulkIn.bSequence != bSequence))
260 {
261 ccid_raise_error("Wrong slot in response");
262 rc = SCARD_ERR(E_READER_UNSUPPORTED);
263 }
264 else
265 {
266 rc = ccid_recv_to_slot_status(packet);
267 if (rc == SCARD_ERR(E_TIMEOUT))
268 {
269 /* This is a time extension */
270 wTimeExtension++;
271 D(printf("Time extension %d...\n", wTimeExtension));
272 if (wTimeExtension <= 120)
273 goto again;
274 /* More than 2 minutes seems too much... */
275 rc = SCARD_ERR(F_WAITED_TOO_LONG);
276 }
277 CCID_LIB(NextSequence)(bSlot);
278 }
279 break;
280
281 case CCID_COMM_INTERRUPT_RDR_TO_PC:
282 /* Interrupts should not arrive here */
283 ccid_raise_error("Received an Interrupt instead of a Response");
284 rc = SCARD_ERR(E_READER_UNSUPPORTED);
285 break;
286
287 default:
288 ccid_raise_error("Unsupported Endpoint");
289 rc = SCARD_ERR(E_READER_UNSUPPORTED);
290 break;
291 }
292
293 return rc;
294}

299LONG CCID_LIB(WaitInterrupt)(CCID_PACKET_ST* packet, DWORD timeout_ms)
300{
301 LONG rc;
302
303 if (packet == NULL)
304 {
305 ccid_raise_error("NULL packet in CCID_WaitInterrupt");
306 return SCARD_ERR(F_INTERNAL_ERROR);
307 }
308
309 rc = CCID_LIB(SerialRecv)(packet, timeout_ms);
310 if (rc != SCARD_ERR(S_SUCCESS))
311 {
312 ccid_raise_error("Failed to receive Interrupt packet from device");
313 return rc;
314 }
315
316 if (packet->bEndpoint != CCID_COMM_INTERRUPT_RDR_TO_PC)
317 {
318 ccid_raise_error("Wrong endpoint for Interrupt");
319 rc = SCARD_ERR(E_READER_UNSUPPORTED);
320 }
321 else if (packet->Header.p.bRequest != RDR_TO_PC_INTERRUPT)
322 {
323 ccid_raise_error("Wrong opcode for Interrupt");
324 rc = SCARD_ERR(E_READER_UNSUPPORTED);
325 }
326
327 return rc;
328}
