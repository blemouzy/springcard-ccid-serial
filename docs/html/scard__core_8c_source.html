76LONG SCARD_LIB(Status)(BYTE bSlot, BOOL* pfCardPresent, BOOL* pfCardPowered)
77{
78 CCID_PACKET_ST packet;
79 LONG rc;
80
81 CCID_LIB(PacketInit)(&packet);
82
83 packet.bEndpoint = CCID_COMM_BULK_PC_TO_RDR;
84 packet.Header.p.bRequest = PC_TO_RDR_GETSLOTSTATUS;
85 packet.Header.p.Data.BulkOut.bSlot = bSlot;
86 packet.Header.p.Data.BulkOut.bSequence = CCID_LIB(GetSequence)(bSlot);
87
88 rc = CCID_LIB(Exchange)(&packet, BULK_TIMEOUT);
89 if (SCARD_LIB(IsFatalError)(rc))
90 return rc; /* Fatal error encountered, no need to go further */
91
92 if (packet.Header.p.bRequest != RDR_TO_PC_SLOTSTATUS)
93 {
94 scard_raise_error("Wrong opcode in response to GET SLOT STATUS");
95 rc = SCARD_ERR(E_READER_UNSUPPORTED);
96 }
97 else
98 {
99 rc = SCARD_ERR(S_SUCCESS);
100 switch (packet.Header.p.Data.BulkIn.bSlotStatus & 0x03)
101 {
102 case 0x00:
103 if (pfCardPresent != NULL)
104 *pfCardPresent = TRUE;
105 if (pfCardPowered != NULL)
106 *pfCardPowered = TRUE;
107 break;
108 case 0x01:
109 if (pfCardPresent != NULL)
110 *pfCardPresent = TRUE;
111 if (pfCardPowered != NULL)
112 *pfCardPowered = FALSE;
113 break;
114 case 0x02:
115 if (pfCardPresent != NULL)
116 *pfCardPresent = FALSE;
117 if (pfCardPowered != NULL)
118 *pfCardPowered = FALSE;
119 break;
120 case 0x03:
121 default:
122 scard_raise_error("Wrong STATUS value in response to GET SLOT STATUS");
123 rc = SCARD_ERR(E_READER_UNSUPPORTED);
124 }
125 }
126
127 return rc;
128}

145LONG SCARD_LIB(Connect)(BYTE bSlot, BYTE abAtr[], DWORD* pdwAtrLength)
146{
147 CCID_PACKET_ST packet;
148 LONG rc;
149
150 if ((abAtr == NULL) || (pdwAtrLength == NULL))
151 return SCARD_ERR(E_INVALID_PARAMETER);
152
153 CCID_LIB(PacketInit)(&packet);
154
155 packet.bEndpoint = CCID_COMM_BULK_PC_TO_RDR;
156 packet.Header.p.bRequest = PC_TO_RDR_ICCPOWERON;
157 packet.Header.p.Data.BulkOut.bSlot = bSlot;
158 packet.Header.p.Data.BulkOut.bSequence = CCID_LIB(GetSequence)(bSlot);
159
160 packet.abRecvPayload = abAtr;
161 packet.dwRecvPayloadMaxLen = *pdwAtrLength;
162
163 rc = CCID_LIB(Exchange)(&packet, BULK_TIMEOUT);
164
165 if (rc == SCARD_ERR(S_SUCCESS))
166 {
167 if (packet.Header.p.bRequest != RDR_TO_PC_DATABLOCK)
168 {
169 rc = SCARD_ERR(E_READER_UNSUPPORTED);
170 }
171 else
172 {
173 *pdwAtrLength = packet.Header.p.Length.dw;
174 }
175 }
176
177 return rc;
178}

230LONG SCARD_LIB(Transmit)(BYTE bSlot, const BYTE abSendApdu[], DWORD dwSendLength, BYTE abRecvApdu[], DWORD *pdwRecvLength)
231{
232 CCID_PACKET_ST packet;
233 LONG rc;
234
235 if (abSendApdu == NULL)
236 return SCARD_ERR(E_INVALID_PARAMETER);
237 if ((abRecvApdu != NULL) && (pdwRecvLength == NULL))
238 return SCARD_ERR(E_INVALID_PARAMETER);
239 if (dwSendLength > CCID_MAX_PAYLOAD_LENGTH)
240 return SCARD_ERR(E_NO_MEMORY);
241
242 CCID_LIB(PacketInit)(&packet);
243
244 packet.bEndpoint = CCID_COMM_BULK_PC_TO_RDR;
245 packet.Header.p.bRequest = PC_TO_RDR_XFRBLOCK;
246 packet.Header.p.Data.BulkOut.bSlot = bSlot;
247 packet.Header.p.Data.BulkOut.bSequence = CCID_LIB(GetSequence)(bSlot);
248
249 packet.abSendPayload = abSendApdu;
250 packet.Header.p.Length.dw = dwSendLength;
251
252 if (pdwRecvLength != NULL)
253 {
254 packet.abRecvPayload = abRecvApdu;
255 packet.dwRecvPayloadMaxLen = *pdwRecvLength;
256 }
257
258 rc = CCID_LIB(Exchange)(&packet, BULK_TIMEOUT);
259
260 if (rc == SCARD_ERR(S_SUCCESS))
261 if (pdwRecvLength != NULL)
262 *pdwRecvLength = packet.Header.p.Length.dw;
263
264 if ((rc == SCARD_ERR(W_UNSUPPORTED_CARD)) ||
265 (rc == SCARD_ERR(W_UNRESPONSIVE_CARD)) ||
266 (rc == SCARD_ERR(W_UNPOWERED_CARD)) ||
267 (rc == SCARD_ERR(W_RESET_CARD)))
268 {
269 rc = SCARD_ERR(W_REMOVED_CARD);
270 }
271
272 return rc;
273}

287LONG SCARD_LIB(Control)(const BYTE abSendBuffer[], DWORD dwSendLength, BYTE abRecvBuffer[], DWORD *pdwRecvLength)
288{
289 CCID_PACKET_ST packet;
290 BYTE bDummyRecvByte;
291 LONG rc;
292
293 if (abSendBuffer == NULL)
294 return SCARD_ERR(E_INVALID_PARAMETER);
295 if ((abRecvBuffer != NULL) && (pdwRecvLength == NULL))
296 return SCARD_ERR(E_INVALID_PARAMETER);
297 if (dwSendLength > CCID_MAX_PAYLOAD_LENGTH)
298 return SCARD_ERR(E_NO_MEMORY);
299
300 CCID_LIB(PacketInit)(&packet);
301
302 packet.bEndpoint = CCID_COMM_BULK_PC_TO_RDR;
303 packet.Header.p.bRequest = PC_TO_RDR_ESCAPE;
304
305 packet.abSendPayload = abSendBuffer;
306 packet.Header.p.Length.dw = dwSendLength;
307
308 if (pdwRecvLength != NULL)
309 {
310 packet.abRecvPayload = abRecvBuffer;
311 packet.dwRecvPayloadMaxLen = *pdwRecvLength;
312 }
313 else
314 {
315 packet.abRecvPayload = &bDummyRecvByte;
316 packet.dwRecvPayloadMaxLen = 1;
317 }
318
319 rc = CCID_LIB(Exchange)(&packet, BULK_TIMEOUT);
320
321 if (!SCARD_LIB(IsFatalError)(rc))
322 {
323 if (packet.Header.p.bRequest != RDR_TO_PC_ESCAPE)
324 {
325 rc = SCARD_ERR(E_READER_UNSUPPORTED);
326 }
327 else
328 {
329 rc = SCARD_ERR(S_SUCCESS);
330 if (pdwRecvLength != NULL)
331 {
332 *pdwRecvLength = packet.Header.p.Length.dw;
333 }
334 else if (bDummyRecvByte != 0)
335 {
336 /* The device has returned an error */
337 return SCARD_ERR(F_UNKNOWN_ERROR);
338 }
339 }
340 }
341
342 return rc;
343}

359LONG SCARD_LIB(GetStatusChangeEx)(DWORD dwTimeoutMs, DWORD* pdwPresentSlots, DWORD* pdwChangedSlots)
360{
361 CCID_PACKET_ST packet;
362 BYTE abInterruptBuffer[CCID_MAX_INTERRUPT_PAYLOAD_LENGTH];
363 LONG rc;
364 DWORD dwPresentSlots = 0;
365 DWORD dwChangedSlots = 0;
366
367 CCID_LIB(PacketInit)(&packet);
368
369 packet.abRecvPayload = abInterruptBuffer;
370 packet.dwRecvPayloadMaxLen = sizeof(abInterruptBuffer);
371
372 rc = CCID_LIB(WaitInterrupt)(&packet, dwTimeoutMs);
373
374 if (rc == SCARD_ERR(S_SUCCESS))
375 {
376 if (packet.Header.p.Length.dw > 1)
377 {
378 if (abInterruptBuffer[0] & 0x01)
379 dwPresentSlots |= 0x00000001;
380 if (abInterruptBuffer[0] & 0x02)
381 dwChangedSlots |= 0x00000001;
382 if (abInterruptBuffer[0] & 0x04)
383 dwPresentSlots |= 0x00000002;
384 if (abInterruptBuffer[0] & 0x08)
385 dwChangedSlots |= 0x00000002;
386 if (abInterruptBuffer[0] & 0x10)
387 dwPresentSlots |= 0x00000004;
388 if (abInterruptBuffer[0] & 0x20)
389 dwChangedSlots |= 0x00000004;
390 if (abInterruptBuffer[0] & 0x40)
391 dwPresentSlots |= 0x00000008;
392 if (abInterruptBuffer[0] & 0x80)
393 dwChangedSlots |= 0x00000008;
394 }
395 if (packet.Header.p.Length.dw > 2)
396 {
397 if (abInterruptBuffer[1] & 0x01)
398 dwPresentSlots |= 0x00000010;
399 if (abInterruptBuffer[1] & 0x02)
400 dwChangedSlots |= 0x00000010;
401 if (abInterruptBuffer[1] & 0x04)
402 dwPresentSlots |= 0x00000020;
403 if (abInterruptBuffer[1] & 0x08)
404 dwChangedSlots |= 0x00000020;
405 if (abInterruptBuffer[1] & 0x10)
406 dwPresentSlots |= 0x00000040;
407 if (abInterruptBuffer[1] & 0x20)
408 dwChangedSlots |= 0x00000040;
409 if (abInterruptBuffer[1] & 0x40)
410 dwPresentSlots |= 0x00000080;
411 if (abInterruptBuffer[1] & 0x80)
412 dwChangedSlots |= 0x00000080;
413 }
414
415 D(printf("Interrupt, slots present: %08lX, slots changed: %08lX\n", dwPresentSlots, dwChangedSlots));
416
417 if (pdwPresentSlots != NULL)
418 *pdwPresentSlots = dwPresentSlots;
419 if (pdwChangedSlots != NULL)
420 *pdwChangedSlots = dwChangedSlots;
421 }
422
423 return rc;
424}
