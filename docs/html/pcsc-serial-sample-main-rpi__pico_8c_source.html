84int main(void)
85{
86 LONG rc;
87
88 platform_init();
89
90 while (1)
91 {
92 int ch = getchar_timeout_us(100);
93 if (ch != PICO_ERROR_TIMEOUT)
94 {
95 execute_command(ch);
96 }
97
98 if (sample_running)
99 {
100 printf("Looking for CCID device through the UART...\n");
101 CCID_LIB(Init)();
102 rc = CCID_LIB(Ping)();
103 if (rc != SCARD_ERR(S_SUCCESS))
104 {
105 printf("No device found on UART (rc=%lX), check hardware and try again\n", rc);
106 }
107 else
108 {
109 printf("CCID device found, starting the sample (use 's' to stop)\n");
110 SCARD_LIB(Init)();
111 sample();
112 if (sample_running)
113 {
114 /* The sample has exited spontaneously */
115 printf("The sample has exited after a fatal error\n");
116 }
117 else
118 {
119 /* User has terminated the sample */
120 printf("Sample stopped (use 'r' to restart)\n");
121 }
122 }
123 sample_running = FALSE;
124 }
125 }
126}

128BOOL SCARD_LIB(IsCancelledHook)(void)
129{
130 /* Keep receiving on USB to be able to stop the sample */
131 int ch = getchar_timeout_us(0);
132 if (ch == 's')
133 {
134 /* Stop --> Cancelled by user */
135 sample_running = FALSE;
136 }
137 return !sample_running;
138}

143static void execute_command(char command)
144{
145 if (sample_running)
146 {
147 switch (command)
148 {
149 case 's' :
150 sample_running = false;
151 break;
152 default:
153 printf("Sample is running (use 's' to stop)\n");
154 }
155 }
156 else
157 {
158 switch (command)
159 {
160 case 'i' :
161 fCcidUseNotifications = !fCcidUseNotifications;
162 printf("Notifications (Interrupt endpoint) are %s\n", fCcidUseNotifications ? ENABLED_STRING : DISABLED_STRING);
163 break;
164 case 'c' :
165 fTestEchoControl = !fTestEchoControl;
166 printf("ECHO test over SCardControl is %s\n", fTestEchoControl ? ENABLED_STRING : DISABLED_STRING);
167 break;
168 case 't' :
169 fTestEchoTransmit = !fTestEchoTransmit;
170 printf("ECHO test over SCardTransmit is %s\n", fTestEchoTransmit ? ENABLED_STRING : DISABLED_STRING);
171 break;
172 case 'v' :
173 fVerbose = !fVerbose;
174 printf("Verbose output is %s\n", fVerbose ? ENABLED_STRING : DISABLED_STRING);
175 break;
176 case 'r':
177 sample_running = true;
178 break;
179 case 's':
180 printf("Sample is not running ('r' to start)\n");
181 break;
182 case 'h' :
183 case '?' :
184 default :
185 print_banner();
186 print_menu();
187 }
188 }
189}

222static void platform_init(void)
223{
224 /* Initialize the libraries */
225 stdio_init_all();
226
227 /* Say hello */
228 printf("This is SpringCard ccid-serial for Raspberry Pico / Raspberry Pico W\n");
229
230 /* For Raspberry Pico W: find the WiFi chip */
231 if (cyw43_arch_init() == 0)
232 {
233 printf("WiFi init success, this must be a Raspberry Pico W\n");
234 has_wifi = true;
235 }
236 else
237 {
238 printf("WiFi init failed, this must be a Raspberry Pico\n");
239 }
240
241 if (!has_wifi)
242 {
243 /* Initialize the hardware for the Raspberry Pico */
244 gpio_init(LED_PIN);
245 gpio_set_dir(LED_PIN, GPIO_OUT);
246 }
247 else
248 {
249 /* TODO: Initialize the hardware for the Raspberry Pico W */
250 }
251
252 /* Preparing... */
253 set_led(true);
254
255 /* Set up the UART */
256 uart_init(UART_ID, BAUD_RATE);
257
258 /* Set the TX and RX pins by using the function select on the GPIO */
259 /* Set datasheet for more information on function select */
260 gpio_set_function(UART_TX_PIN, GPIO_FUNC_UART);
261 gpio_set_function(UART_RX_PIN, GPIO_FUNC_UART);
262
263 /* Confirm the baudrate */
264 uart_set_baudrate(UART_ID, BAUD_RATE);
265
266 /* Set UART flow control CTS/RTS, we don't want these, so turn them off */
267 uart_set_hw_flow(UART_ID, false, false);
268
269 /* Set our data format */
270 uart_set_format(UART_ID, DATA_BITS, STOP_BITS, PARITY);
271
272 /* Turn off FIFO's - we want to do this character by character */
273 uart_set_fifo_enabled(UART_ID, false);
274
275 /* Set up a RX interrupt */
276 /* We need to set up the handler first */
277 /* Select correct interrupt for the UART we are using */
278 int UART_IRQ = UART_ID == uart0 ? UART0_IRQ : UART1_IRQ;
279
280 /* And set up and enable the interrupt handlers */
281 irq_set_exclusive_handler(UART_IRQ, on_uart_rx);
282 irq_set_enabled(UART_IRQ, true);
283
284 /* Now enable the UART to send interrupts - RX only */
285 uart_set_irq_enables(UART_ID, true, false);
286
287 /* Ready... */
288 set_led(false);
289}

291static void set_led(bool onoff)
292{

296 cyw43_arch_gpio_put(CYW43_WL_GPIO_LED_PIN, 0);

299 }
300 else
301 {
302 if (has_wifi)
303 cyw43_arch_gpio_put(CYW43_WL_GPIO_LED_PIN, 1);
304 else
305 gpio_put(LED_PIN, 1);
306 }
