85BOOL SCARD_LIB(IsValidContext)(void)
86{
87 if (scard_valid)
88 {
89 /* Hook to be able to change the status (not used on all targets) */
90 if (SCARD_LIB(IsCancelledHook)())
91 {
92 printf("PC/SC-Like context not valid, operation has been cancelled by the user\n");
93 scard_valid = FALSE;
94 }
95 if (!CCID_LIB(IsValidDriver)())
96 {
97 printf("PC/SC-Like context not valid, the CCID driver has reported an error\n");
98 scard_valid = FALSE;
99 }
100 if (!CCID_LIB(SerialIsOpen)())
101 {
102 printf("PC/SC-Like context not valid, CCID serial port is not open\n");
103 scard_valid = FALSE;
104 }
105 }
106 return scard_valid;
107}

112BOOL SCARD_LIB(IsFatalError)(LONG rc)
113{
114 switch (rc)
115 {
116 case SCARD_ERR(S_SUCCESS):
117 case SCARD_ERR(W_UNSUPPORTED_CARD):
118 case SCARD_ERR(W_UNRESPONSIVE_CARD):
119 case SCARD_ERR(W_UNPOWERED_CARD):
120 case SCARD_ERR(W_RESET_CARD):
121 case SCARD_ERR(W_REMOVED_CARD):
122 case SCARD_ERR(W_INSERTED_CARD):
123 case SCARD_ERR(E_NO_SMARTCARD):
124 case SCARD_ERR(E_SHARING_VIOLATION):
125 case SCARD_ERR(E_PROTO_MISMATCH):
126 case SCARD_ERR(E_UNKNOWN_CARD):
127 case SCARD_ERR(E_INVALID_ATR):
128 return FALSE;
129
130 default:
131 return TRUE;
132 }
133}
